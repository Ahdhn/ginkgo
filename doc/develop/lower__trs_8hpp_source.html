245  explicit LowerTrs(const Factory *factory,
246  const LowerTrsArgs<ValueType> &args)
247  : parameters_{factory->get_parameters()},
248  EnableLinOp<LowerTrs>(factory->get_executor(),
249  transpose(args.system_matrix->get_size())),
250  b_{std::move(args.b)},
251  system_matrix_{}
252  {
253  using CsrMatrix = matrix::Csr<ValueType, IndexType>;
254 
255  GKO_ASSERT_IS_SQUARE_MATRIX(args.system_matrix);
256  // This is needed because it does not make sense to call the copy and
257  // convert if the existing matrix is empty.
258  const auto exec = this->get_executor();
259  if (!args.system_matrix->get_size()) {
260  system_matrix_ = CsrMatrix::create(exec);
261  } else {
262  system_matrix_ =
263  copy_and_convert_to<CsrMatrix>(exec, args.system_matrix);
264  }
265  if (parameters_.preconditioner) {
266  preconditioner_ =
267  parameters_.preconditioner->generate(system_matrix_);
268  } else {
269  preconditioner_ = matrix::Identity<ValueType>::create(
270  this->get_executor(), this->get_size()[0]);
271  }
272  this->generate();
273  }
