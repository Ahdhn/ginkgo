560  template <typename T>
561  void copy_from(const Executor *src_exec, size_type num_elems,
562  const T *src_ptr, T *dest_ptr) const
563  {
564  this->template log<log::Logger::copy_started>(
565  src_exec, this, reinterpret_cast<uintptr>(src_ptr),
566  reinterpret_cast<uintptr>(dest_ptr), num_elems * sizeof(T));
567  try {
568  this->raw_copy_from(src_exec, num_elems * sizeof(T), src_ptr,
569  dest_ptr);
570  } catch (NotSupported &) {
571  // Unoptimized copy. Try to go through the masters.
572  auto src_master = src_exec->get_master().get();
573  if (num_elems > 0 && src_master != src_exec) {
574  auto *master_ptr = src_exec->get_master()->alloc<T>(num_elems);
575  src_master->copy_from<T>(src_exec, num_elems, src_ptr,
576  master_ptr);
577  this->copy_from<T>(src_master, num_elems, master_ptr, dest_ptr);
578  src_master->free(master_ptr);
579  }
580  }
581  this->template log<log::Logger::copy_completed>(
582  src_exec, this, reinterpret_cast<uintptr>(src_ptr),
583  reinterpret_cast<uintptr>(dest_ptr), num_elems * sizeof(T));
584  }
