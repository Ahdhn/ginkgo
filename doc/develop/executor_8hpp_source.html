636 class ExecutorBase : public Executor {
637 public:
638  void run(const Operation &op) const override
639  {
640  this->template log<log::Logger::operation_launched>(this, &op);
641  op.run(self()->shared_from_this());
642  this->template log<log::Logger::operation_completed>(this, &op);
643  }
644 
645 protected:
646  void raw_copy_from(const Executor *src_exec, size_type n_bytes,
647  const void *src_ptr, void *dest_ptr) const override
648  {
649  src_exec->raw_copy_to(self(), n_bytes, src_ptr, dest_ptr);
650  }
651 
652 private:
653  ConcreteExecutor *self() noexcept
654  {
655  return static_cast<ConcreteExecutor *>(this);
656  }
657

659  {
660  return static_cast<const ConcreteExecutor *>(this);
661  }
