564  template <typename T>
565  void copy_from(const Executor *src_exec, size_type num_elems,
566  const T *src_ptr, T *dest_ptr) const
567  {
568  this->template log<log::Logger::copy_started>(
569  src_exec, this, reinterpret_cast<uintptr>(src_ptr),
570  reinterpret_cast<uintptr>(dest_ptr), num_elems * sizeof(T));
571  try {
572  this->raw_copy_from(src_exec, num_elems * sizeof(T), src_ptr,
573  dest_ptr);
574  } catch (NotSupported &) {
575  // Unoptimized copy. Try to go through the masters.
576  auto src_master = src_exec->get_master().get();
577  if (num_elems > 0 && src_master != src_exec) {
578  auto *master_ptr = src_exec->get_master()->alloc<T>(num_elems);
579  src_master->copy_from<T>(src_exec, num_elems, src_ptr,
580  master_ptr);
581  this->copy_from<T>(src_master, num_elems, master_ptr, dest_ptr);
582  src_master->free(master_ptr);
583  }
584  }
585  this->template log<log::Logger::copy_completed>(
586  src_exec, this, reinterpret_cast<uintptr>(src_ptr),
587  reinterpret_cast<uintptr>(dest_ptr), num_elems * sizeof(T));
588  }
