561  template <typename T>
562  void copy_from(const Executor *src_exec, size_type num_elems,
563  const T *src_ptr, T *dest_ptr) const
564  {
565  this->template log<log::Logger::copy_started>(
566  src_exec, this, reinterpret_cast<uintptr>(src_ptr),
567  reinterpret_cast<uintptr>(dest_ptr), num_elems * sizeof(T));
568  try {
569  this->raw_copy_from(src_exec, num_elems * sizeof(T), src_ptr,
570  dest_ptr);
571  } catch (NotSupported &) {
572  // Unoptimized copy. Try to go through the masters.
573  auto src_master = src_exec->get_master().get();
574  if (num_elems > 0 && src_master != src_exec) {
575  auto *master_ptr = src_exec->get_master()->alloc<T>(num_elems);
576  src_master->copy_from<T>(src_exec, num_elems, src_ptr,
577  master_ptr);
578  this->copy_from<T>(src_master, num_elems, master_ptr, dest_ptr);
579  src_master->free(master_ptr);
580  }
581  }
582  this->template log<log::Logger::copy_completed>(
583  src_exec, this, reinterpret_cast<uintptr>(src_ptr),
584  reinterpret_cast<uintptr>(dest_ptr), num_elems * sizeof(T));
585  }
