88 template <typename IndexType>
89 struct block_interleaved_storage_scheme {
93  IndexType block_offset;
97  IndexType group_offset;
103  uint32 group_power;
104 
110  GKO_ATTRIBUTES IndexType get_group_size() const noexcept
111  {
112  return one<IndexType>() << group_power;
113  }
114 
123  GKO_ATTRIBUTES IndexType compute_storage_space(IndexType num_blocks) const
124  noexcept
125  {
126  return (num_blocks + 1 == size_type{0})
127  ? size_type{0}
128  : ceildiv(num_blocks, this->get_group_size()) * group_offset;
129  }
130 
138  GKO_ATTRIBUTES IndexType get_group_offset(IndexType block_id) const noexcept
139  {
140  return group_offset * (block_id >> group_power);
141  }
142 
150  GKO_ATTRIBUTES IndexType get_block_offset(IndexType block_id) const noexcept
151  {
152  return block_offset * (block_id & (this->get_group_size() - 1));
153  }
154 
162  GKO_ATTRIBUTES IndexType get_global_block_offset(IndexType block_id) const
163  noexcept
164  {
165  return this->get_group_offset(block_id) +
166  this->get_block_offset(block_id);
167  }
168 
174  GKO_ATTRIBUTES IndexType get_stride() const noexcept
175  {
176  return block_offset << group_power;
177  }
178 };

318 
326  static size_type get_larger_power(size_type n) noexcept
327  {
328  size_type res = 1;
329  while (res < n) res *= 2;
330  return res;
331  }
332 
336  static uint32 get_log2(size_type n) noexcept
337  {
338  for (auto r = uint32{0};; ++r) {
339  if ((size_type{1} << (r + 1)) > n) {
340  return r;
341  }
342  }
343  }
344 
353  static block_interleaved_storage_scheme<index_type> compute_storage_scheme(
354  uint32 max_block_size) noexcept
355  {
356  const auto group_size = static_cast<uint32>(
357  max_block_stride_ / get_larger_power(max_block_size));
358  const auto block_offset = max_block_size;
359  const auto block_stride = group_size * block_offset;
360  const auto group_offset = max_block_size * block_stride;
361  return {static_cast<index_type>(block_offset),
362  static_cast<index_type>(group_offset), get_log2(group_size)};
363  }
