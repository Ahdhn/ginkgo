95  GKO_ATTRIBUTES size_type compute_storage_space(size_type num_blocks) const
96  noexcept
97  {
98  return (num_blocks + 1 == size_type{0})
99  ? size_type{0}
100  : ceildiv(num_blocks, this->get_group_size()) * group_offset;

111  {

122  GKO_ATTRIBUTES IndexType get_block_offset(IndexType block_id) const noexcept
123  {
124  return block_offset * (block_id & (this->get_group_size() - 1));
125  }

134  GKO_ATTRIBUTES IndexType get_global_block_offset(IndexType block_id) const
135  noexcept

138  this->get_block_offset(block_id);
139  }

146  GKO_ATTRIBUTES IndexType get_stride() const noexcept
147  {

450  static block_interleaved_storage_scheme<index_type> compute_storage_scheme(
451  uint32 max_block_size) noexcept
452  {
453  const auto group_size = static_cast<uint32>(
454  max_block_stride_ / get_superior_power(uint32{2}, max_block_size));
455  const auto block_offset = max_block_size;
456  const auto block_stride = group_size * block_offset;
457  const auto group_offset = max_block_size * block_stride;
458  return {static_cast<index_type>(block_offset),
459  static_cast<index_type>(group_offset),
460  get_significant_bit(group_size)};
461  }
