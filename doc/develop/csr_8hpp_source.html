158  void process(const Array<index_type> &mtx_row_ptrs,
159  Array<index_type> *mtx_srow)
160  {
161  constexpr uint32 warp_size = 32;
162  auto nwarps = mtx_srow->get_num_elems();
163 
164  if (nwarps > 0) {
165  auto exec = mtx_srow->get_executor()->get_master();
166  Array<index_type> srow_host(exec);
167  srow_host = *mtx_srow;
168  auto srow = srow_host.get_data();
169  Array<index_type> row_ptrs_host(exec);
170  row_ptrs_host = mtx_row_ptrs;
171  auto row_ptrs = row_ptrs_host.get_const_data();
172  for (size_type i = 0; i < nwarps; i++) {
173  srow[i] = 0;
174  }
175  auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
176  auto num_elems = row_ptrs[num_rows];
177  for (size_type i = 0; i < num_rows; i++) {
178  auto bucket =
179  ceildiv((ceildiv(row_ptrs[i + 1], warp_size) * nwarps),
180  ceildiv(num_elems, warp_size));
181  if (bucket < nwarps) {
182  srow[bucket]++;
183  }
184  }
185  // find starting row for thread i
186  for (size_type i = 1; i < nwarps; i++) {
187  srow[i] += srow[i - 1];
188  }
189  *mtx_srow = srow_host;
190  }
191  }

193  int64_t clac_size(const int64_t nnz)
194  {
195  constexpr uint32 warp_size = 32;
196  int multiple = 8;
197  if (nnz >= 2000000) {
198  multiple = 128;
199  } else if (nnz >= 200000) {
200  multiple = 32;
201  }
202  auto nwarps = nwarps_ * multiple;
203  return min(ceildiv(nnz, warp_size), static_cast<int64_t>(nwarps));
204  }
205 
206  private:
207  int64_t nwarps_;
208  };

225  void process(const Array<index_type> &mtx_row_ptrs,
226  Array<index_type> *mtx_srow)
227  {
228  // if the number of stored elements per row is larger than 1e6 or
229  // the maximum number of stored elements per row is larger than
230  // 64, use load_balance otherwise use classical
231  const auto num = mtx_row_ptrs.get_num_elems();
232  Array<index_type> host_row_ptrs(
233  mtx_row_ptrs.get_executor()->get_master());
234  host_row_ptrs = mtx_row_ptrs;
235  const auto row_val = host_row_ptrs.get_const_data();
236  if (row_val[num] > 1e6) {
237  std::make_shared<load_balance>(nwarps_)->process(host_row_ptrs,
238  mtx_srow);
239  this->set_name("load_balance");
240  } else {
241  index_type maxnum = 0;
242  for (index_type i = 1; i < num; i++) {
243  maxnum = max(maxnum, row_val[i] - row_val[i - 1]);
244  }
245  if (maxnum > 64) {
246  std::make_shared<load_balance>(nwarps_)->process(
247  host_row_ptrs, mtx_srow);
248  this->set_name("load_balance");
249  } else {
250  std::make_shared<classical>()->process(host_row_ptrs,
251  mtx_srow);
252  this->set_name("classical");
253  }
254  }
255  }
256 
257  int64_t clac_size(const int64_t nnz)
258  {
259  return std::make_shared<load_balance>(nwarps_)->clac_size(nnz);
260  }
