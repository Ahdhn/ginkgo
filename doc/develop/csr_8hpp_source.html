325  void process(const Array<index_type> &mtx_row_ptrs,
326  Array<index_type> *mtx_srow) override
327  {
328  auto nwarps = mtx_srow->get_num_elems();
329 
330  if (nwarps > 0) {
331  auto host_srow_exec = mtx_srow->get_executor()->get_master();
332  auto host_mtx_exec = mtx_row_ptrs.get_executor()->get_master();
333  const bool is_srow_on_host{host_srow_exec ==
334  mtx_srow->get_executor()};
335  const bool is_mtx_on_host{host_mtx_exec ==
336  mtx_row_ptrs.get_executor()};
337  Array<index_type> row_ptrs_host(host_mtx_exec);
338  Array<index_type> srow_host(host_srow_exec);
339  const index_type *row_ptrs{};
340  index_type *srow{};
341  if (is_srow_on_host) {
342  srow = mtx_srow->get_data();
343  } else {
344  srow_host = *mtx_srow;
345  srow = srow_host.get_data();
346  }
347  if (is_mtx_on_host) {
348  row_ptrs = mtx_row_ptrs.get_const_data();
349  } else {
350  row_ptrs_host = mtx_row_ptrs;
351  row_ptrs = row_ptrs_host.get_const_data();
352  }
353  for (size_type i = 0; i < nwarps; i++) {
354  srow[i] = 0;
355  }
356  const auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
357  const auto num_elems = row_ptrs[num_rows];
358  for (size_type i = 0; i < num_rows; i++) {
359  auto bucket =
360  ceildiv((ceildiv(row_ptrs[i + 1], warp_size_) * nwarps),
361  ceildiv(num_elems, warp_size_));
362  if (bucket < nwarps) {
363  srow[bucket]++;
364  }
365  }
366  // find starting row for thread i
367  for (size_type i = 1; i < nwarps; i++) {
368  srow[i] += srow[i - 1];
369  }
370  if (!is_srow_on_host) {
371  *mtx_srow = srow_host;
372  }
373  }
374  }

377  {
378  if (warp_size_ > 0) {
379  int multiple = 8;
380  if (nnz >= 2e6) {
381  multiple = 128;
382  } else if (nnz >= 2e5) {
383  multiple = 32;
384  }
385 
386 #if GINKGO_HIP_PLATFORM_HCC
387  if (!cuda_strategy_) {
388  multiple = 8;

390  multiple = 64;
391  } else if (nnz >= 1e6) {
392  multiple = 16;
393  }
394  }
395 #endif // GINKGO_HIP_PLATFORM_HCC
396 
397  auto nwarps = nwarps_ * multiple;
398  return min(ceildiv(nnz, warp_size_), int64_t(nwarps));
399  } else {
400  return 0;

458  void process(const Array<index_type> &mtx_row_ptrs,
459  Array<index_type> *mtx_srow) override
460  {
461  // if the number of stored elements is larger than <nnz_limit> or
462  // the maximum number of stored elements per row is larger than
463  // <row_len_limit>, use load_balance otherwise use classical
464  // CUDA: nnz_limit = 1e6, row_len_limit = 1024
465  // AMD: nnz_limit = 1e8, row_len_limit = 768
466  index_type nnz_limit = 1e6;
467  index_type row_len_limit = 1024;
468 #if GINKGO_HIP_PLATFORM_HCC
469  if (!cuda_strategy_) {
470  nnz_limit = 1e8;
471  row_len_limit = 768;
472  }
473 #endif // GINKGO_HIP_PLATFORM_HCC
474  auto host_mtx_exec = mtx_row_ptrs.get_executor()->get_master();
475  const bool is_mtx_on_host{host_mtx_exec ==
476  mtx_row_ptrs.get_executor()};
477  Array<index_type> row_ptrs_host(host_mtx_exec);
478  const index_type *row_ptrs{};
479  if (is_mtx_on_host) {
480  row_ptrs = mtx_row_ptrs.get_const_data();
481  } else {
482  row_ptrs_host = mtx_row_ptrs;
483  row_ptrs = row_ptrs_host.get_const_data();
484  }
485  const auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
486  if (row_ptrs[num_rows] > nnz_limit) {
487  load_balance actual_strategy(nwarps_, warp_size_,
488  cuda_strategy_);
489  if (is_mtx_on_host) {
490  actual_strategy.process(mtx_row_ptrs, mtx_srow);
491  } else {
492  actual_strategy.process(row_ptrs_host, mtx_srow);
493  }
494  this->set_name(actual_strategy.get_name());
495  } else {
496  index_type maxnum = 0;
497  for (index_type i = 1; i < num_rows + 1; i++) {
498  maxnum = max(maxnum, row_ptrs[i] - row_ptrs[i - 1]);
499  }
500  if (maxnum > row_len_limit) {
501  load_balance actual_strategy(nwarps_, warp_size_,
502  cuda_strategy_);
503  if (is_mtx_on_host) {
504  actual_strategy.process(mtx_row_ptrs, mtx_srow);
505  } else {
506  actual_strategy.process(row_ptrs_host, mtx_srow);
507  }
508  this->set_name(actual_strategy.get_name());
509  } else {
510  classical actual_strategy;
511  if (is_mtx_on_host) {
512  actual_strategy.process(mtx_row_ptrs, mtx_srow);
513  max_length_per_row_ =
514  actual_strategy.get_max_length_per_row();
515  } else {
516  actual_strategy.process(row_ptrs_host, mtx_srow);
517  max_length_per_row_ =
518  actual_strategy.get_max_length_per_row();
519  }
520  this->set_name(actual_strategy.get_name());
521  }
522  }

836 template <typename ValueType, typename IndexType>
837 void strategy_rebuild_helper(Csr<ValueType, IndexType> *result)
838 {
839  using load_balance = typename Csr<ValueType, IndexType>::load_balance;
840  using automatical = typename Csr<ValueType, IndexType>::automatical;
841  auto strategy = result->get_strategy();
842  auto executor = result->get_executor();
843  if (std::dynamic_pointer_cast<load_balance>(strategy)) {
844  if (auto exec =
845  std::dynamic_pointer_cast<const HipExecutor>(executor)) {
846  result->set_strategy(std::make_shared<load_balance>(exec));
847  } else if (auto exec = std::dynamic_pointer_cast<const CudaExecutor>(
848  executor)) {
849  result->set_strategy(std::make_shared<load_balance>(exec));
850  }
851  } else if (std::dynamic_pointer_cast<automatical>(strategy)) {
852  if (auto exec =
853  std::dynamic_pointer_cast<const HipExecutor>(executor)) {
854  result->set_strategy(std::make_shared<automatical>(exec));
855  } else if (auto exec = std::dynamic_pointer_cast<const CudaExecutor>(
856  executor)) {
857  result->set_strategy(std::make_shared<automatical>(exec));
858  }
859  }
860 }
