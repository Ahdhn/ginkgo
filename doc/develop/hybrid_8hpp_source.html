115  void compute_hybrid_config(const Array<size_type> &row_nnz,
116  size_type *ell_num_stored_elements_per_row,
117  size_type *coo_nnz)
118  {
119  Array<size_type> ref_row_nnz(row_nnz.get_executor()->get_master(),
120  row_nnz.get_num_elems());
121  ref_row_nnz = row_nnz;
122  ell_num_stored_elements_per_row_ =
123  this->compute_ell_num_stored_elements_per_row(&ref_row_nnz);
124  coo_nnz_ = this->compute_coo_nnz(ref_row_nnz);
125  *ell_num_stored_elements_per_row = ell_num_stored_elements_per_row_;
126  *coo_nnz = coo_nnz_;
127  }

165  size_type compute_coo_nnz(const Array<size_type> &row_nnz) const
166  {
167  size_type coo_nnz = 0;
168  auto row_nnz_val = row_nnz.get_const_data();
169  for (size_type i = 0; i < row_nnz.get_num_elems(); i++) {
170  if (row_nnz_val[i] > ell_num_stored_elements_per_row_) {
171  coo_nnz +=
172  row_nnz_val[i] - ell_num_stored_elements_per_row_;
173  }
174  }
175  return coo_nnz;
176  }
177 
178  private:
179  size_type ell_num_stored_elements_per_row_;
180  size_type coo_nnz_;

215  class imbalance_limit : public strategy_type {
216  public:
223  explicit imbalance_limit(float percent = 0.8) : percent_(percent)
224  {
225  percent_ = std::min(percent_, 1.0f);
226  percent_ = std::max(percent_, 0.0f);
227  }
228 
229  size_type compute_ell_num_stored_elements_per_row(
230  Array<size_type> *row_nnz) const override
231  {
232  auto row_nnz_val = row_nnz->get_data();
233  auto num_rows = row_nnz->get_num_elems();
234  std::sort(row_nnz_val, row_nnz_val + num_rows);
235  if (percent_ < 1) {
236  auto percent_pos = static_cast<size_type>(num_rows * percent_);
237  return row_nnz_val[percent_pos];
238  } else {
239  return row_nnz_val[num_rows - 1];
240  }
241  }
242 
243  private:
244  float percent_;
245  };
