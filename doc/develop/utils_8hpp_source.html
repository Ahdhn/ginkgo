417 template <typename T>
418 class temporary_clone {
419 public:
420  using value_type = T;
421  using pointer = T *;
422 
429  explicit temporary_clone(std::shared_ptr<const Executor> exec, pointer ptr)
430  {
431  if (ptr->get_executor() == exec) {
432  // just use the object we already have
433  handle_ = handle_type(ptr, [](pointer) {});
434  } else {
435  // clone the object to the new executor and make sure it's copied
436  // back before we delete it
437  handle_ = handle_type(gko::clone(std::move(exec), ptr).release(),
438  copy_back_deleter<T>(ptr));
439  }
440  }
441 
447  T *get() const { return handle_.get(); }
448 
454  T *operator->() const { return handle_.get(); }
455 
456 private:
457  // std::function deleter allows to decide the (type of) deleter at runtime
458  using handle_type = std::unique_ptr<T, std::function<void(T *)>>;
459 
460  handle_type handle_;
461 };
