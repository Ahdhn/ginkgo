90 template <typename ValueType, typename AccuracyType, typename CondType,
91  typename Predicate1, typename Predicate2>
92 GKO_ATTRIBUTES precision_reduction
93 get_optimal_storage_reduction(AccuracyType accuracy, CondType cond,
94  Predicate1 verificator1, Predicate2 verificator2)
95 {
96  using gko::detail::float_traits;
97  using type = remove_complex<ValueType>;
98  auto accurate = [&cond, &accuracy](double eps) {
99  return cond * eps < accuracy;
100  };
101  uint8 is_verified1 = 2;
102  // the following code uses short-circuiting to avoid calling possibly
103  // expensive verificatiors multiple times
104  if (accurate(float_traits<truncate_type<truncate_type<type>>>::eps)) {
105  return precision_reduction(2, 0);
106  } else if (accurate(
107  float_traits<truncate_type<reduce_precision<type>>>::eps) &&
108  (is_verified1 = verificator1())) {
109  return precision_reduction(1, 1);
110  } else if (accurate(float_traits<
111  reduce_precision<reduce_precision<type>>>::eps) &&
112  is_verified1 != 0 && verificator2()) {
113  return precision_reduction(0, 2);
114  } else if (accurate(float_traits<truncate_type<type>>::eps)) {
115  return precision_reduction(1, 0);
116  } else if (accurate(float_traits<reduce_precision<type>>::eps) &&
117  (is_verified1 == 1 ||
118  (is_verified1 == 2 && (is_verified1 = verificator1())))) {
119  return precision_reduction(0, 1);
120  } else {
121  return precision_reduction(0, 0);
122  }
123 }
