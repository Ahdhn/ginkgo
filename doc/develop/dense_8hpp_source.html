384 template <typename Matrix, typename... TArgs>
385 std::unique_ptr<Matrix> initialize(
386  size_type stride, std::initializer_list<typename Matrix::value_type> vals,
387  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
388 {
389  using dense = matrix::Dense<typename Matrix::value_type>;
390  size_type num_rows = vals.size();
391  auto tmp = dense::create(exec->get_master(), dim{num_rows, 1}, stride);
392  size_type idx = 0;
393  for (const auto &elem : vals) {
394  tmp->at(idx) = elem;
395  ++idx;
396  }
397  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
398  tmp->move_to(mtx.get());
399  return mtx;
400 }

448 template <typename Matrix, typename... TArgs>
449 std::unique_ptr<Matrix> initialize(
450  size_type stride,
451  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
452  vals,
453  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
454 {
455  using dense = matrix::Dense<typename Matrix::value_type>;
456  size_type num_rows = vals.size();
457  size_type num_cols = num_rows > 0 ? begin(vals)->size() : 1;
458  auto tmp =
459  dense::create(exec->get_master(), dim{num_rows, num_cols}, stride);
460  size_type ridx = 0;
461  for (const auto &row : vals) {
462  size_type cidx = 0;
463  for (const auto &elem : row) {
464  tmp->at(ridx, cidx) = elem;
465  ++cidx;
466  }
467  ++ridx;
468  }
469  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
470  tmp->move_to(mtx.get());
471  return mtx;
472 }
