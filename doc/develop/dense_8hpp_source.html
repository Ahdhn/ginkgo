453 template <typename Matrix, typename... TArgs>
454 std::unique_ptr<Matrix> initialize(
455  size_type stride, std::initializer_list<typename Matrix::value_type> vals,
456  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
457 {
458  using dense = matrix::Dense<typename Matrix::value_type>;
459  size_type num_rows = vals.size();
460  auto tmp = dense::create(exec->get_master(), dim<2>{num_rows, 1}, stride);
461  size_type idx = 0;
462  for (const auto &elem : vals) {
463  tmp->at(idx) = elem;
464  ++idx;
465  }
466  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
467  tmp->move_to(mtx.get());
468  return mtx;
469 }

517 template <typename Matrix, typename... TArgs>
518 std::unique_ptr<Matrix> initialize(
519  size_type stride,
520  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
521  vals,
522  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
523 {
524  using dense = matrix::Dense<typename Matrix::value_type>;

527  auto tmp =
528  dense::create(exec->get_master(), dim<2>{num_rows, num_cols}, stride);
529  size_type ridx = 0;
530  for (const auto &row : vals) {
531  size_type cidx = 0;
532  for (const auto &elem : row) {
533  tmp->at(ridx, cidx) = elem;
534  ++cidx;
535  }
536  ++ridx;
537  }
538  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
539  tmp->move_to(mtx.get());
