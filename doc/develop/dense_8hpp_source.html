573 template <typename Matrix, typename... TArgs>
574 std::unique_ptr<Matrix> initialize(
575  size_type stride, std::initializer_list<typename Matrix::value_type> vals,
576  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
577 {
578  using dense = matrix::Dense<typename Matrix::value_type>;
579  size_type num_rows = vals.size();
580  auto tmp = dense::create(exec->get_master(), dim<2>{num_rows, 1}, stride);
581  size_type idx = 0;
582  for (const auto &elem : vals) {
583  tmp->at(idx) = elem;
584  ++idx;
585  }
586  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
587  tmp->move_to(mtx.get());
588  return mtx;
589 }

643 template <typename Matrix, typename... TArgs>
644 std::unique_ptr<Matrix> initialize(
645  size_type stride,
646  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
647  vals,
648  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
649 {
650  using dense = matrix::Dense<typename Matrix::value_type>;
651  size_type num_rows = vals.size();
652  size_type num_cols = num_rows > 0 ? begin(vals)->size() : 1;
653  auto tmp =
654  dense::create(exec->get_master(), dim<2>{num_rows, num_cols}, stride);
655  size_type ridx = 0;
656  for (const auto &row : vals) {
657  size_type cidx = 0;
658  for (const auto &elem : row) {
659  tmp->at(ridx, cidx) = elem;
660  ++cidx;
661  }

663  }
664  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
665  tmp->move_to(mtx.get());
