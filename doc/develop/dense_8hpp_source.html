341 template <typename Matrix, typename... TArgs>
342 std::unique_ptr<Matrix> initialize(
343  size_type stride, std::initializer_list<typename Matrix::value_type> vals,
344  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
345 {
346  using dense = matrix::Dense<typename Matrix::value_type>;
347  size_type num_rows = vals.size();
348  auto tmp = dense::create(exec->get_master(), dim{num_rows, 1}, stride);
349  size_type idx = 0;
350  for (const auto &elem : vals) {
351  tmp->at(idx) = elem;
352  ++idx;
353  }
354  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
355  tmp->move_to(mtx.get());
356  return mtx;
357 }

405 template <typename Matrix, typename... TArgs>
406 std::unique_ptr<Matrix> initialize(
407  size_type stride,
408  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
409  vals,
410  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
411 {
412  using dense = matrix::Dense<typename Matrix::value_type>;
413  size_type num_rows = vals.size();
414  size_type num_cols = num_rows > 0 ? begin(vals)->size() : 1;
415  auto tmp =
416  dense::create(exec->get_master(), dim{num_rows, num_cols}, stride);
417  size_type ridx = 0;
418  for (const auto &row : vals) {
419  size_type cidx = 0;
420  for (const auto &elem : row) {
421  tmp->at(ridx, cidx) = elem;
422  ++cidx;
423  }
424  ++ridx;
425  }
426  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
427  tmp->move_to(mtx.get());
428  return mtx;
