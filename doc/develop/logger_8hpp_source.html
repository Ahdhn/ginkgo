118 #define GKO_LOGGER_REGISTER_EVENT(_id, _event_name, ...) \
119 protected: \
120  virtual void on_##_event_name(__VA_ARGS__) const {} \
121  \
122 public: \
123  template <size_type Event, typename... Params> \
124  xstd::enable_if_t<Event == _id && (_id < event_count_max)> on( \
125  Params &&... params) const \
126  { \
127  if (enabled_events_ & (mask_type{1} << _id)) { \
128  this->on_##_event_name(std::forward<Params>(params)...); \
129  } \
130  } \
131  static constexpr size_type _event_name{_id}; \
132  static constexpr mask_type _event_name##_mask{mask_type{1} << _id};
