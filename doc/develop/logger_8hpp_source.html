103 #define GKO_LOGGER_REGISTER_EVENT(_id, _event_name, ...) \
104 protected: \
105  virtual void on_##_event_name(__VA_ARGS__) const {} \
106  \
107 public: \
108  template <size_type Event, typename... Params> \
109  xstd::enable_if_t<Event == _id && (_id < event_count_max)> on( \
110  Params &&... params) const \
111  { \
112  if (enabled_events_ & (mask_type{1} << _id)) { \
113  this->on_##_event_name(std::forward<Params>(params)...); \
114  } \
115  } \
116  static constexpr size_type _event_name{_id}; \
117  static constexpr mask_type _event_name##_mask{mask_type{1} << _id};
