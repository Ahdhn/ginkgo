514 template <typename ConcreteLoggable, typename PolymorphicBase = Loggable>
515 class EnableLogging : public Loggable {
516 public:
517  void add_logger(std::shared_ptr<const Logger> logger) override
518  {
519  loggers_.push_back(logger);
520  }
521 
522  void remove_logger(const Logger *logger) override
523  {
524  auto idx = find_if(begin(loggers_), end(loggers_),
525  [&logger](std::shared_ptr<const Logger> l) {
526  return lend(l) == logger;
527  });
528  if (idx != end(loggers_)) {
529  loggers_.erase(idx);
530  } else {
531  throw OutOfBoundsError(__FILE__, __LINE__, loggers_.size(),
532  loggers_.size());
533  }
534  }
535 
536 protected:
537  template <size_type Event, typename... Params>
538  void log(Params &&... params) const
539  {
540  for (auto &logger : loggers_) {
541  logger->template on<Event>(std::forward<Params>(params)...);
542  }
543  }
544 
545  std::vector<std::shared_ptr<const Logger>> loggers_;
546 };
