234 class precision_reduction {
235 public:
239  using storage_type = uint8;
240 
241 private:
242  static constexpr auto nonpreserving_bits = 4u;
243  static constexpr auto preserving_bits =
244  byte_size * sizeof(storage_type) - nonpreserving_bits;
245  static constexpr auto nonpreserving_mask =
246  storage_type{(0x1 << nonpreserving_bits) - 1};
247  static constexpr auto preserving_mask =
248  storage_type{(0x1 << preserving_bits) - 1} << nonpreserving_bits;
249 
250 public:
256  GKO_ATTRIBUTES constexpr precision_reduction() noexcept : data_{0x0} {}
257 
265  GKO_ATTRIBUTES constexpr precision_reduction(
266  storage_type preserving, storage_type nonpreserving) noexcept
267  : data_((GKO_ASSERT(preserving < (0x1 << preserving_bits) - 1),
268  GKO_ASSERT(nonpreserving < (0x1 << nonpreserving_bits) - 1),
269  (preserving << nonpreserving_bits) | nonpreserving))
270  {}
