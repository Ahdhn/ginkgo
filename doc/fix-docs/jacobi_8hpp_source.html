509  static block_interleaved_storage_scheme<index_type> compute_storage_scheme(
510  uint32 max_block_size) noexcept
511  {
512  const auto group_size = static_cast<uint32>(
513  max_block_stride_ / get_superior_power(uint32{2}, max_block_size));
514  const auto block_offset = max_block_size;
515  const auto block_stride = group_size * block_offset;
516  const auto group_offset = max_block_size * block_stride;
517  return {static_cast<index_type>(block_offset),
518  static_cast<index_type>(group_offset),
519  get_significant_bit(group_size)};
520  }
