356  std::unique_ptr<Dense> create_submatrix(const span &rows,

360  row_major_range range_this{this->get_values(), this->get_size()[0],

519 template <typename Matrix, typename... TArgs>
520 std::unique_ptr<Matrix> initialize(
521  size_type stride, std::initializer_list<typename Matrix::value_type> vals,
522  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
523 {
524  using dense = matrix::Dense<typename Matrix::value_type>;
525  size_type num_rows = vals.size();
526  auto tmp = dense::create(exec->get_master(), dim<2>{num_rows, 1}, stride);
527  size_type idx = 0;
528  for (const auto &elem : vals) {
529  tmp->at(idx) = elem;
530  ++idx;
531  }
532  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
533  tmp->move_to(mtx.get());
534  return mtx;
535 }

589 template <typename Matrix, typename... TArgs>
590 std::unique_ptr<Matrix> initialize(
591  size_type stride,
592  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
593  vals,
594  std::shared_ptr<const Executor> exec, TArgs &&... create_args)
595 {
596  using dense = matrix::Dense<typename Matrix::value_type>;
597  size_type num_rows = vals.size();
598  size_type num_cols = num_rows > 0 ? begin(vals)->size() : 1;
599  auto tmp =
600  dense::create(exec->get_master(), dim<2>{num_rows, num_cols}, stride);
601  size_type ridx = 0;
602  for (const auto &row : vals) {
603  size_type cidx = 0;
604  for (const auto &elem : row) {
605  tmp->at(ridx, cidx) = elem;
606  ++cidx;
607  }
608  ++ridx;
609  }
610  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
611  tmp->move_to(mtx.get());
612  return mtx;
613 }
