236  size_type compute_ell_num_stored_elements_per_row(
237  Array<size_type> *row_nnz) const override
238  {
239  auto row_nnz_val = row_nnz->get_data();
240  auto num_rows = row_nnz->get_num_elems();
241  std::sort(row_nnz_val, row_nnz_val + num_rows);
242  if (percent_ < 1) {
243  auto percent_pos = static_cast<size_type>(num_rows * percent_);
244  return row_nnz_val[percent_pos];
245  } else {
246  return row_nnz_val[num_rows - 1];
247  }
248  }
249 
250  private:
251  float percent_;
252  };
