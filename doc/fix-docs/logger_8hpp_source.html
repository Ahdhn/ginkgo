521 template <typename ConcreteLoggable, typename PolymorphicBase = Loggable>
522 class EnableLogging : public Loggable {
523 public:
524  void add_logger(std::shared_ptr<const Logger> logger) override
525  {
526  loggers_.push_back(logger);
527  }
528 
529  void remove_logger(const Logger *logger) override
530  {
531  auto idx = find_if(begin(loggers_), end(loggers_),
532  [&logger](std::shared_ptr<const Logger> l) {
533  return lend(l) == logger;
534  });
535  if (idx != end(loggers_)) {
536  loggers_.erase(idx);
537  } else {
538  throw OutOfBoundsError(__FILE__, __LINE__, loggers_.size(),
539  loggers_.size());
540  }
541  }
542 
543 protected:
544  template <size_type Event, typename... Params>
545  void log(Params &&... params) const
546  {
547  for (auto &logger : loggers_) {
548  logger->template on<Event>(std::forward<Params>(params)...);
549  }
550  }
