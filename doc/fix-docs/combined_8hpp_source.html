116 template <typename FactoryContainer>
117 std::shared_ptr<const CriterionFactory> combine(FactoryContainer &&factories)
118 {
119  switch (factories.size()) {
120  case 0:
121  GKO_NOT_SUPPORTED(nullptr);
122  return nullptr;
123  case 1:
124  return factories[0];
125  default:
126  auto exec = factories[0]->get_executor();
127  return Combined::build()
128  .with_criteria(std::forward<FactoryContainer>(factories))
129  .on(exec);
130  }
