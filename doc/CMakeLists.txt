cmake_minimum_required(VERSION 2.8)
include(helpers.cmake)
find_package(Doxygen REQUIRED)
option(DOC_GENERATE_PDF "Generate PDF documentation" ON)
option(DOC_GENERATE_DEV "Generate internal documentation" OFF)
if (DOC_GENERATE_PDF)
  find_package(LATEX COMPONENTS PDFLATEX REQUIRED)
endif()

#TODO set version
set(VERSION_MAJOR 1)
set(VERSION_MINOR 0)
doc_gen(usr Doxyfile-usr.in OFF)
if (DOC_GENERATE_DEV)
  doc_gen(dev Doxyfile-dev.in OFF)
endif()
if (DOC_GENERATE_PDF)
  doc_gen(usr_pdf Doxyfile-usr.in ON)
endif()
