cmake_minimum_required(VERSION 2.8)
include(helpers.cmake)
find_package(Doxygen REQUIRED)
option(DOC_GENERATE_PDF "Generate PDF documentation" OFF)
option(DOC_GENERATE_DEV "Generate internal documentation" OFF)
if (DOC_GENERATE_PDF)
  find_package(LATEX COMPONENTS PDFLATEX REQUIRED)
endif()

doc_gen(usr Doxyfile-usr.in OFF DOCUMENTATION.md)
if (DOC_GENERATE_DEV)
  doc_gen(dev Doxyfile-dev.in OFF DEVELOPER.md)
endif()
if (DOC_GENERATE_PDF)
  doc_gen(pdf Doxyfile-pdf.in ON WHITEPAPER.md)
endif()
